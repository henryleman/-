import requests webhook_url = "https://discord.com/api/webhooks/...." # Dán URL của bạn vào đây message = { "content": "📈 Xin chào! Boy đã gửi tin nhắn thành công từ Discord Webhook 🚀" } response = requests.post(webhook_url, json=message) if response.status_code == 204: print("✅ Gửi thành công!") else: print(f"❌ Lỗi gửi: {response.status_code}")
